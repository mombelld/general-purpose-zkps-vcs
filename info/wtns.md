# Binary format for witness

### Format of the file

````

     ┏━━━━┳━━━━━━━━━━━━━━━━━━━━━┓
     ┃ 4  │   119 116 110 116   ┃     Magic  "wtns"
     ┗━━━━┻━━━━━━━━━━━━━━━━━━━━━┛
     ┏━━━━┳━━━━━━━━━━━━━━━━━━━━━┓
     ┃ 4  │     01 00 00 00     ┃       Version 1
     ┗━━━━┻━━━━━━━━━━━━━━━━━━━━━┛
     ┏━━━━┳━━━━━━━━━━━━━━━━━━━━━┓
     ┃ 4  │     02 00 00 00     ┃       Number of Sections
     ┗━━━━┻━━━━━━━━━━━━━━━━━━━━━┛
     ┏━━━━┳━━━━━━━━━━━━━━━━━┳━━━━━┳━━━━━━━━━━━━━━━━━━━━━━━━┓
     ┃ 4  │ sectionType     ┃  8  │   SectionSize          ┃
     ┗━━━━┻━━━━━━━━━━━━━━━━━┻━━━━━┻━━━━━━━━━━━━━━━━━━━━━━━━┛
     ┏━━━━┳━━━━━━━━━━━━━━━━━━━━━┓
     ┃ 4  │     fieldSize (fs)  ┃       
     ┗━━━━┻━━━━━━━━━━━━━━━━━━━━━┛
     ┏━━━━┳━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┓
     ┃ fs │     prime                         ┃       
     ┗━━━━┻━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛
     ┏━━━━┳━━━━━━━━━━━━━━━━━━━━━┓
     ┃ 4  │     witnessSize     ┃       
     ┗━━━━┻━━━━━━━━━━━━━━━━━━━━━┛

     ┏━━━━┳━━━━━━━━━━━━━━━━━┳━━━━━┳━━━━━━━━━━━━━━━━━━━━━━━━┓
     ┃ 4  │ sectionType     ┃  8  │   SectionSize          ┃
     ┗━━━━┻━━━━━━━━━━━━━━━━━┻━━━━━┻━━━━━━━━━━━━━━━━━━━━━━━━┛
     ┏━━━━┳━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┓  \
     ┃ fs │     w_0                           ┃   |     
     ┗━━━━┻━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛   |
     ...                                           \
     ...                                           / witnessSize
     ┏━━━━┳━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┓   |
     ┃ fs │     w_{witnessSize - 1}           ┃   |    
     ┗━━━━┻━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━┛  /

````
